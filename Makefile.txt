# Define the compiler
CC = gcc

# Define compiler flags
CFLAGS = -Wall -g

# Define the target executable
TARGET = interface

# Define source and object files
SOURCES = interface.c
OBJECTS = $(SOURCES:.c=.o)

# Default target
all: $(TARGET)

# Rule to build the target executable
$(TARGET): $(OBJECTS)
    $(CC) -o $(TARGET) $(OBJECTS)

# Rule to build object files
%.o: %.c
    $(CC) $(CFLAGS) -c $< -o $@

# Rule to clean up build files
clean:
    rm -f $(OBJECTS) $(TARGET)
